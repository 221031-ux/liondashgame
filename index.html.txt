    <style>
        body { margin: 0; overflow: hidden; background-color: #87CEEB; }
        #ui {
            position: absolute; top: 20px; left: 20px;
            color: white; font-family: sans-serif; font-size: 24px;
            font-weight: bold; text-shadow: 2px 2px 4px #000;
        }
        #gameOver {
            position: absolute; top: 50%; left: 50%;
            transform: translate(-50%, -50%);
            color: red; font-family: sans-serif; font-size: 48px;
            font-weight: bold; text-shadow: 2px 2px 4px #000;
            display: none; text-align: center;
        }
    </style>
    <script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r128/three.min.js"></script>

<body>

    <div id="ui">Score: <span id="score">0</span></div>
    <div id="gameOver">GAME OVER<br><span style="font-size: 24px;">Refresh to Restart</span></div>

    <script>
        // --- Game Setup ---
        const scene = new THREE.Scene();
        scene.fog = new THREE.Fog(0x87CEEB, 10, 50); // Adds fog for distance fading

        const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 100);
        camera.position.set(0, 5, 8);
        camera.lookAt(0, 0, -10);

        const renderer = new THREE.WebGLRenderer({ antialias: true });
        renderer.setSize(window.innerWidth, window.innerHeight);
        document.body.appendChild(renderer.domElement);

        // --- Lighting ---
        const light = new THREE.DirectionalLight(0xffffff, 1);
        light.position.set(0, 10, 10);
        scene.add(light);
        const ambientLight = new THREE.AmbientLight(0x404040);
        scene.add(ambientLight);

        // --- Environment (Ground) ---
        const groundGeo = new THREE.PlaneGeometry(20, 200);
        const groundMat = new THREE.MeshLambertMaterial({ color: 0x8B4513 }); // Brown dirt/street
        const ground = new THREE.Mesh(groundGeo, groundMat);
        ground.rotation.x = -Math.PI / 2;
        ground.position.z = -50;
        scene.add(ground);

        // --- Player (Lion Dancer Placeholder) ---
        // Using a Red/Gold cube to represent the lion head
        const playerGeo = new THREE.BoxGeometry(1, 1, 1);
        const playerMat = new THREE.MeshLambertMaterial({ color: 0xFF0000 }); // Red
        const player = new THREE.Mesh(playerGeo, playerMat);
        scene.add(player);

        // --- Game Variables ---
        const lanes = [-2.5, 0, 2.5]; // Left, Center, Right lanes
        let currentLane = 1; // Starts in center
        player.position.set(lanes[currentLane], 0.5, 0);

        let isJumping = false;
        let yVelocity = 0;
        let gravity = -0.015;
        let jumpStrength = 0.3;

        let obstacles = [];
        let speed = 0.3;
        let score = 0;
        let isGameOver = false;

        // --- Controls ---
        window.addEventListener('keydown', (e) => {
            if (isGameOver) return;
            if (e.key === 'ArrowLeft' && currentLane > 0) {
                currentLane--;
                player.position.x = lanes[currentLane];
            }
            if (e.key === 'ArrowRight' && currentLane < 2) {
                currentLane++;
                player.position.x = lanes[currentLane];
            }
            if (e.key === 'ArrowUp' && !isJumping) {
                isJumping = true;
                yVelocity = jumpStrength;
            }
        });

        // --- Obstacle Generation ---
        function spawnObstacle() {
            if (isGameOver) return;
            const obsGeo = new THREE.BoxGeometry(1.5, 1.5, 1.5);
            const obsMat = new THREE.MeshLambertMaterial({ color: 0x808080 }); // Grey barriers
            const obstacle = new THREE.Mesh(obsGeo, obsMat);
            
            // Random lane
            const randomLane = Math.floor(Math.random() * 3);
            obstacle.position.set(lanes[randomLane], 0.75, -60);
            
            scene.add(obstacle);
            obstacles.push(obstacle);

            // Spawn next obstacle
            setTimeout(spawnObstacle, Math.random() * 1000 + 500); // 0.5s to 1.5s
        }
        spawnObstacle();

        // --- Game Loop ---
        function animate() {
            if (isGameOver) return;
            requestAnimationFrame(animate);

            // Jump Physics
            if (isJumping) {
                player.position.y += yVelocity;
                yVelocity += gravity;
                if (player.position.y <= 0.5) {
                    player.position.y = 0.5;
                    isJumping = false;
                }
            }

            // Move Obstacles & Check Collisions
            for (let i = obstacles.length - 1; i >= 0; i--) {
                let obs = obstacles[i];
                obs.position.z += speed;

                // Collision detection (Simple bounding box overlap)
                if (
                    Math.abs(player.position.x - obs.position.x) < 1.0 &&
                    Math.abs(player.position.y - obs.position.y) < 1.0 &&
                    Math.abs(player.position.z - obs.position.z) < 1.0
                ) {
                    isGameOver = true;
                    document.getElementById('gameOver').style.display = 'block';
                }

                // Remove passed obstacles and increase score
                if (obs.position.z > 5) {
                    scene.remove(obs);
                    obstacles.splice(i, 1);
                    score += 10;
                    document.getElementById('score').innerText = score;
                    speed += 0.001; // Slowly increase difficulty
                }
            }

            renderer.render(scene, camera);
        }
        animate();

        // Resize handler
        window.addEventListener('resize', () => {
            renderer.setSize(window.innerWidth, window.innerHeight);
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
        });
    </script>
